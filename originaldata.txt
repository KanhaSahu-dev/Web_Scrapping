    document.getElementById('clickMe').addEventListener('click', () => {
      const message = {
        name: document.getElementById('name').value,
        page: document.getElementById('pageno').value // include the value of the input field
      }; // Object to be sent to the server
      // Send POST request to server when button is clicked
      fetch('http://localhost:3000/print', {
        method: 'POST',
        headers: {
          'Content-Type': 'text/plain' // Specify JSON content type
        },
        body: JSON.stringify(message) // Convert object to JSON string
      })
      .then(response => {
        if (response.ok) {
          console.log('Request sent to server');
        } else {
          console.error('Error:', response.statusText);
        }
      })
      .catch(error => {
        console.error('Error:', error);
      });
    });
